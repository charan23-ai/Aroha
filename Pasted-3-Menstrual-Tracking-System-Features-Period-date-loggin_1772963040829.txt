/lib/menstrualEngine.ts
export function predictNextPeriod(lastPeriod:string, cycleLength=28){

  const start = new Date(lastPeriod)
  const next = new Date(start)

  next.setDate(start.getDate()+cycleLength)

  return next.toDateString()
}
/app/track/period/page.tsx
"use client"

import { useState } from "react"
import { predictNextPeriod } from "@/lib/menstrualEngine"

export default function PeriodTracker(){

  const [last,setLast]=useState("")
  const [next,setNext]=useState("")

  const calculate=()=>{
    setNext(predictNextPeriod(last))
  }

  return(

    <div className="p-6">

      <h1 className="text-xl mb-4">
        Period Tracker 🌸
      </h1>

      <input
      type="date"
      onChange={(e)=>setLast(e.target.value)}
      className="border p-3 rounded-xl w-full mb-4"
      />

      <button
      onClick={calculate}
      className="bg-primary text-white w-full py-3 rounded-xl"
      >
      Predict Next Cycle
      </button>

      {next && (
        <p className="mt-4">
          Estimated next cycle: {next}
        </p>
      )}

    </div>
  )
}

7 📊 Emotional Pattern Insights for Cycles
/lib/cycleInsights.ts
export function analyzeCyclePatterns(moodData){

if(moodData.length<3){
return "More cycles needed to detect patterns."
}

const irritability = moodData.filter(m=>m.mood<3).length

if(irritability>3){
return "Mood dips appear during menstrual phase. Hormonal shifts can influence emotions."
}

return "Your mood stays fairly stable across cycles."
}
